# Діаграма прецедентів
## Загальна схема
@startuml
  actor Користувач
  actor Редактор
  actor Адміністратор
  
  usecase Реєстрація
  usecase Авторизація
  
  usecase "Редагування існуючих \nданих" as РІД
  usecase "Управління даними \nсистемами" as УДС
  usecase "Управління редакторами \nсистеми" as УРС
  usecase "Взаємодія з \nданими" as ВЗД


  
  
  Редактор -u-|> Користувач
  Адміністратор -u-|> Редактор
  
  Реєстрація <- Користувач
  Користувач -u-> Авторизація
  Користувач -> ВЗД

  
  Редактор -l-> РІД

  УДС <- Адміністратор
  Адміністратор -> УРС
@enduml

## Схема користувача

@startuml
  actor Користувач
  
  
  usecase Реєстрація
  usecase Авторизація
  
  usecase "Взаємодія з \nданими" as ВЗД
  usecase "Запит на пошук \nданих" as ЗНПД
  usecase "Запит на редагування \nданих" as ЗНРД



  
  

  Реєстрація <- Користувач
  Користувач -u-> Авторизація
  Користувач -> ВЗД
  
  ЗНПД -u-> ВЗД
  ВЗД <-u- ЗНРД


  

@enduml


## Схема Редактора
@startuml
  actor Редактор
  
  usecase "Редагування розмітки \nфайлу" as РРФ
  
  Редактор -u-> РРФ

@enduml

## Схема адміністратора

@startuml
  actor Адміністратор
  
  usecase "Видалення \nфайлу" as ВФ
  usecase "Зміна розмітки \nфайлу" as ЗРФ
  usecase "Історія дій над \nфайлом" as ІДНФ
  usecase "Завантаження нових \nданих" as ЗНД
  usecase "Управління даними \nсистеми" as УДС
  usecase "Надання користувачу прав \nредактора" as НКПР
  usecase "Гілка змін \nредактора" as ГЗР
  usecase "Позбавлення користувача \nправ редактора" as ПКПР
  usecase "Управління редакторами \nсистеми" as УРС

  
  Адміністратор -u-> УДС
  ВФ --> УДС
  ЗРФ --> УДС
  ІДНФ --> УДС
  ЗНД --> УДС
  Адміністратор --> УРС
  НКПР --> УРС
  ГЗР --> УРС
  ПКПР --> УРС


@enduml

| ------------------------ | ------------------------------------------------------------------------------ |
| **_НАЗВА:_**             |  Створення облікового запису користувача                                                 |
| **_УЧАСНИКИ:_**          | Користувач, Система                                                           |
| **_ПЕРЕДУМОВИ:_**        | Відсутність облікового запису користувача                         |
| **_РЕЗУЛЬТАТ:_**         | Запит на змінення маркування файлу, надісланий адміністратору                  |
| **_ВИКЛЮЧНІ СИТУАЦІЇ:_** | 1. Некоректно введені дані для реєстрації (USER_DATA_ERROR)       |
|                          | 2. Обліковий запис користувача вже існує (ACC_EXIST_ERROR)                              |
 **_ОСНОВНИЙ СЦЕНАРІЙ:_**

@startuml

|Користувач|
start;
: Користувач починає роботу з продуктом;
: Користувач розпочинає реєстрацію;
: Користувач вводить необхідні дані;

|Система|
: Система перевіряє коректність введених даних
<font color="red"><b> USER_DATA_ERROR;
: Система перевіряє наявність зареєстрованого облікового запису
<font color="red"><b> ACC_EXIST_ERROR;
: Система створює новий обліковий запис користувача;

|Користувач|
: Користувач отримує повідомлення про успішну реєстрацію;

@enduml
